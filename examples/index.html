Timer

const timelyMeth = () => {
 let incr = 0;
 for (const i = 1; i <= 1e6; i++) {
  incr += i;
 }
}
logger.time(timelyMeth);

Custom prefix

import Loggout from 'loggout';
const loggout = new Loggout({
 level: Loggout.LEVELS.debug,
 prefix: level => `LOGG(${level}):`,
});
loggout.error('test');
